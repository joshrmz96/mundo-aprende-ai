async function loadLesson(lessonId) { try { const response = await fetch(`https://api.example.com/lessons/${lessonId}`); if (!response.ok) { throw new Error(`HTTP error! Status: ${response.status}`); } const lesson = await response.json(); // Process and display the lesson displayLesson(lesson); } catch (error) { console.error('Error fetching lesson:', error); // Fallback for offline or error state displayOfflineMessage(); } }
